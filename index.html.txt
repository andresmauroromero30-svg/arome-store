<script>

let carrito=[];

function agregar(nombre){

carrito.push(nombre);

render();
}

function render(){

let lista=document.getElementById("lista");

lista.innerHTML="";

carrito.forEach(item=>{

lista.innerHTML += `<li>${item}</li>`;

});

let total=0;

if(carrito.length>=3){

let grupos=Math.floor(carrito.length/3);

let resto=carrito.length%3;

total=(grupos*100)+(resto*50);

}else{

total=carrito.length*50;

}

document.getElementById("total").innerHTML=
"Total: S/"+total;

}

function enviarWhatsapp(){

let nombre=document.getElementById("nombre").value;
let celular=document.getElementById("celular").value;
let distrito=document.getElementById("distrito").value;

let total=document.getElementById("total").innerText;

let mensaje=
`Hola AROME%0A%0A`+
`Nombre: ${nombre}%0A`+
`Celular: ${celular}%0A`+
`Distrito: ${distrito}%0A%0A`+
`Perfumes:%0A${carrito.join('%0A')}%0A%0A`+
`${total}`;

window.open(
`https://wa.me/51932667883?text=${mensaje}`,
'_blank'
);

}

document.getElementById("buscar")
.addEventListener("keyup",function(){

let filtro=this.value.toLowerCase();

let cards=document.querySelectorAll(".card");

cards.forEach(card=>{

let texto=card.innerText.toLowerCase();

card.style.display=
texto.includes(filtro)
? "block"
: "none";

});

});
